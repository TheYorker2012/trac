549        static function DaysBetweenTimestamps($FirstTimestamp, $SecondTimestamp)
550        {

552                $day_of_year_of_first = (int)date('z',$FirstTimestamp);
553                $day_of_year_of_second = (int)date('z',$SecondTimestamp);

555                if ($difference < 0) {

558                        // dif = dif + diy(1)
559                        $difference += 365 + (int)date('L',$FirstTimestamp);
560                }

564        /// Find the difference between academic times.
565        static function Difference($First, $Second, $Units)
566        {
567                $timestamp = $First->Timestamp();
568                $end = $Second->Timestamp();
569                $result = array();
570                foreach ($Units as $unit) {
571                        // If we've landed on the end date, just return the current result.
572                        if ($timestamp < $end) {
573                                $count = -1;
574                                $inc = $timestamp;
575                                while ($inc <= $end) {
576                                        $timestamp = $inc;
577                                        $inc = strtotime("+1$unit", $timestamp);
578                                        ++$count;
579                                }
580                                $result[$unit] = $count;
581                        } else {
582                                $result[$unit] = 0;
583                        }
584                }
585                return $result;
586        }
