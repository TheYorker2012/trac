36/// Find whether the set of permissions for a level is a subset of another.
37/**
38 * This is analagous to @a $Subset <= @a $Superset.
39 * @param $Subset string Permission level to search for.
40 * @param $Superset string Permission level to search in.
41 * @return bool Whether all permissions of @a $Subset are permissions of
42 *      @a $Superset.
43 */
44function PermissionsSubset($Subset, $Superset)
45{
46        // Tree of which subsets of each permission level.
47        static $permission_subsets = array(
48                'public'                => array(),
49                'student'               => array('public'       => TRUE),
50                'organisation'  => array('student'      => TRUE),
51                'vip'                   => array('student'      => TRUE),
52                'office'                => array('student'      => TRUE),
53                'pr'                    => array('office'       => TRUE),
54                'moderator'             => array('pr'           => TRUE),
55                'editor'                => array('moderator'=> TRUE),
56                'admin'                 => array('editor'       => TRUE),
57        );
58
59        // Unknown superset, assume its empty, always fail.
60        if (!array_key_exists($Superset, $permission_subsets)) {
61                return FALSE;
62        }
63        // If the superset IS the subset allow.
64        if ($Superset === $Subset) {
65                return TRUE;
66        }
67        // If the subset is an explicit subset identifier of superset use it directly.
68        if (array_key_exists($Subset, $permission_subsets[$Superset])) {
69                return $permission_subsets[$Superset][$Subset];
70        }
71        // Go through the supersets subsets in a depth first manner to find subset.
72        foreach ($permission_subsets[$Superset] as $superset_subset=>$enable) {
73                $subset_found = PermissionsSubset($Subset, $superset_subset);
74                if ($subset_found) {
75                        // $Subset is a subset of $Superset
76                        return $enable;
77                }
78                // Negative: try any other parent permissions before giving up.
79        }
80        // All explicit subsets checked. Must not be a subset.
81        return FALSE;
82}

815/// Get the redirect data.
816/**
817 * @return Stored data or NULL on failure.
818 */
819function GetRedirectData()
820{
821        $CI = & get_instance();
822        $PageId = $CI->uri->uri_string();
823        static $result = NULL;
824        if (NULL === $result) {
825                if (array_key_exists('posts',$_SESSION)) {
826                        if (array_key_exists($PageId,$_SESSION['posts'])) {
827                                $found = FALSE;
828                                while (!$found && !empty($_SESSION['posts'][$PageId])) {
829                                        $post_data = array_shift($_SESSION['posts'][$PageId]);
830                                        // Check not expired
831                                        if ($post_data[0] > time()) {
832                                                $result = $post_data[1];
833                                                $found = TRUE;
834                                        }
835                                }
836                                if (empty($_SESSION['posts'][$PageId])) {
837                                        unset($_SESSION['posts'][$PageId]);
838                                }
839                        }
840                }
841        }
842        return $result;
843}
