479function toggleRolePermission(role, permission)
480{
481        var rolePermissionsDiv = document.getElementById('role-'+role+'-permissions');
482        var name = 'permission-'+permission;
483       
484        if (undefined != rolePermissions[role][permission]) {
485                delete rolePermissions[role][permission];
486                delete permissionRoles[permission][role];
487

492                                        children[i].parentNode.removeChild(children[i]);
493                                }
494                        }
495                }
496        }
497        else {

499                permissionRoles[permission][role] = true;

509                        rolePermissionsDiv.appendChild(document.createTextNode(' '));
510                }
511        }
