134// Cycle the options in a select so that first_value is first.
135function CycleSelect(select, first_value, first)
136{
137        var options = select.getElementsByTagName('option');
138        var first_index = -1;
139        var old_end_index = -1;
140       
141        // Check that first_value exists
142        // Also find which one end selected (assume they're at regular intervals)
143        var finish = 0;
144        for (var i = 0; i < options.length && finish <2; ++i) {
145                if (options[i].value == first_value) {
146                        first_index = i;
147                        ++finish;
148                }
149                if (options[i].value == select.value) {
150                        old_end_index = i;
151                        ++finish;
152                }
153        }
154        // Only reorder if first_value exists and not already in correct order
155        if (first_index > 0) {
156                // Each time, remove and reappend the first option
157                for (var i = 0; i < first_index; ++i) {
158                        var first = options[0];
159                        select.removeChild(first);
160                        select.appendChild(first);
161                }
162                // If we know which end was selected, use the same index
163                if (old_end_index>=0) {
164                        options = select.getElementsByTagName('option');
165                        for (var i = 0; i < options.length; ++i) {
166                                if (i == old_end_index) {
167                                        select.value = options[i].value;
168                                        break;
169                                }
170                        }
171                }
172        }
173}

436function MiniTog(date)

438        // If the main edit stuff is hidden (for confirmation) check before doing anything

449                if (1) {

454                                        // inc -> none
455                                        RemoveInexDate('in', date);
456                                } else if (!exc_dates[date]) {
457                                        // none -> exc
458                                        NewInexDate('ex', date);

462                                }
463                        } else {
464                                // deselected, so exc -> none -> inc
465                                if (exc_dates[date]) {
466                                        // exc -> none
467                                        RemoveInexDate('ex', date);
468                                } else if (!inc_dates[date]) {
469                                        // none -> inc
470                                        NewInexDate('in', date);
471                                } else {
472                                        // inic -> none
473                                        RemoveInexDate('in', date);
474                                }
475                        }
476                } else {
477                        // Toggle all three values
478                        if (inc_dates[date]) {
479                                // inc -> exc

485                        } else {
486                                // none -> inc
487                                NewInexDate('in', date);

489                }

491        return false;
