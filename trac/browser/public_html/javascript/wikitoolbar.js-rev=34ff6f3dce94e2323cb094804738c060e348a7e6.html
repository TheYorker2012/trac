182// apply tagOpen/tagClose to selection in textarea,
183// use sampleText instead of selection if there is none
184// copied and adapted from phpBB
185function insertTags(tagOpen, tagClose, sampleText, textarea) {
186
187        var txtarea = document.getElementById(textarea);
188
189        // IE
190        if (document.selection  && !is_gecko) {
191                var theSelection = document.selection.createRange().text;
192                if (!theSelection) {
193                        theSelection=sampleText;
194                }
195                txtarea.focus();
196                if (theSelection.charAt(theSelection.length - 1) == " ") { // exclude ending space char, if any
197                        theSelection = theSelection.substring(0, theSelection.length - 1);
198                        document.selection.createRange().text = tagOpen + theSelection + tagClose + " ";
199                } else {
200                        document.selection.createRange().text = tagOpen + theSelection + tagClose;
201                }
202
203        // Mozilla
204        } else if(txtarea.selectionStart || txtarea.selectionStart == '0') {
205                var replaced = false;
206                var startPos = txtarea.selectionStart;
207                var endPos = txtarea.selectionEnd;
208                if (endPos-startPos) {
209                        replaced = true;
210                }
211                var scrollTop = txtarea.scrollTop;
212                var myText = (txtarea.value).substring(startPos, endPos);
213                if (!myText) {
214                        myText=sampleText;
215                }
216                var subst;
217                if (myText.charAt(myText.length - 1) == " ") { // exclude ending space char, if any
218                        subst = tagOpen + myText.substring(0, (myText.length - 1)) + tagClose + " ";
219                } else {
220                        subst = tagOpen + myText + tagClose;
221                }
222                txtarea.value = txtarea.value.substring(0, startPos) + subst +
223                        txtarea.value.substring(endPos, txtarea.value.length);
224                txtarea.focus();
225                //set new selection
226                if (replaced) {
227                        var cPos = startPos+(tagOpen.length+myText.length+tagClose.length);
228                        txtarea.selectionStart = cPos;
229                        txtarea.selectionEnd = cPos;
230                } else {
231                        txtarea.selectionStart = startPos+tagOpen.length;
232                        txtarea.selectionEnd = startPos+tagOpen.length+myText.length;
233                }
234                txtarea.scrollTop = scrollTop;
235
236        // All other browsers get no toolbar.
237        // There was previously support for a crippled "help"
238        // bar, but that caused more problems than it solved.
239        }
240        // reposition cursor if possible
241        if (txtarea.createTextRange) {
242                txtarea.caretPos = document.selection.createRange().duplicate();
243        }
