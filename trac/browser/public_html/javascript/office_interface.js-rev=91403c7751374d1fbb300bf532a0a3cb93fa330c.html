402function detectPhotoChanges () {
403        var newPhotoData = Object.clone(oi_photoData);
404        var change = false;
405
406        for (var x = 0; x < oi_photoData.length; x++) {
407                // Caption
408                if ($('photoCaption' + oi_photoData[x]['request_id']).value != oi_photoData[x]['photo_caption']) {
409                        newPhotoData[x]['photo_caption'] = $('photoCaption' + oi_photoData[x]['request_id']).value;
410                        change = true;
411                }
412                // ALT
413                if ($('photoALT' + oi_photoData[x]['request_id']).value != oi_photoData[x]['photo_alt']) {
414                        newPhotoData[x]['photo_alt'] = $('photoALT' + oi_photoData[x]['request_id']).value;
415                        change = true;
416                }
417                // Size
418                var size_input = $('photoSize' + oi_photoData[x]['request_id']);
419                if (size_input.options[size_input.selectedIndex].value != oi_photoData[x]['photo_type']) {
420                        newPhotoData[x]['photo_type'] = size_input.options[size_input.selectedIndex].value;
421                        change = true;
422                }
423        }
424
425        if (change) {
426                return newPhotoData;
427        } else {
428                return null;
429        }
430}

432function switchPage (page_id) {
433        if ((!oi_popupShown) && (document.getElementById('nav_' + page_id).className != 'disabled')) {
434                /* Change selected menu option */
435                var nav = document.getElementById('office_nav');
436                for (var i = 0; i < nav.childNodes.length; i++) {
437                        if (nav.childNodes[i].className != 'disabled') {
438                                nav.childNodes[i].className = '';
439                        }
440                }
441                document.getElementById('nav_' + page_id).className = 'current';
442                /* Change content displayed */
443                var canvas = document.getElementById('office_pages');
444                for (var i = 0; i < canvas.childNodes.length; i++) {
445                        if (canvas.childNodes[i].className == 'office_page') {
446                                canvas.childNodes[i].style.display = 'none';
447                        }
448                }
449                document.getElementById('page_' + page_id).style.display = 'block';
450        }
451        return false;
452}
