23CREATE TABLE `notices` (
24 `notice_id` INT NOT NULL AUTO_INCREMENT,
25 `notice_source_entity_id` INT NOT NULL COMMENT 'entity the notice is from',
26 `notice_subject` VARCHAR( 255 ) NOT NULL ,
27 `notice_content_wikitext` MEDIUMTEXT NOT NULL COMMENT 'wikitext content',
28 `notice_content_cache` MEDIUMTEXT NOT NULL COMMENT 'cached xhtml content',
29 `notice_private` TINYINT( 1 ) NOT NULL COMMENT 'whether the message is unaccessible by the public/others',
30 `notice_published` TINYINT( 1 ) NOT NULL COMMENT 'whether the notice has been published yet',
31 `notice_deleted` TINYINT( 1 ) NOT NULL COMMENT 'whether the notice has been deleted',
32 `notice_updated` TIMESTAMP ON UPDATE CURRENT_TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'time last updated or published',
33 `notice_expires` TIMESTAMP NOT NULL COMMENT 'time when the notice is set to expire',
34 PRIMARY KEY ( `notice_id` )
35) ENGINE = MYISAM  COMMENT = 'General purpose notices between entities (particularly to teams/organisations)';
36
37CREATE TABLE `notice_recipients` (
38 `notice_recipient_notice_id` INT NOT NULL COMMENT 'identifies the notice',
39 `notice_recipient_destination_entity_id` INT NOT NULL COMMENT 'entity to send it to',
40 `notice_recipient_blind` TINYINT( 1 ) NOT NULL DEFAULT '0' COMMENT 'whether to allow other recipients to see this recipient',
41 `notice_recipient_sent` TINYINT( 1 ) NOT NULL DEFAULT '0' COMMENT 'whether an email has been sent for this notice',
42 `notice_recipient_dismissed` TINYINT( 1 ) NOT NULL DEFAULT '0' COMMENT 'whether the user has dismissed the notice',
43 PRIMARY KEY ( `notice_recipient_notice_id` , `notice_recipient_destination_entity_id` )
44) ENGINE = MYISAM  COMMENT = 'Recipients of the notices';

46view columns
47        notice_expires <= NOW()                                                 AS      notice_expired
48        notice_published AND NOT notice_deleted                 AS      notice_available
49        notice_available AND NOT notice_expired AND NOT notice_recipient_dismissed                      AS      notice_visible
